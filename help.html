Sample code

The following code implements a binary search tree and uses it to sort a list of integers. 

procedure insert node, item
  if node = nothing
    pass
  else
    if item < node@1
      if node@2 = nothing
        node@2 := [item, nothing, nothing]
      else
        insert node@2, item
    else
      if node@3 = nothing
        node@3 := [item, nothing, nothing]
      else
        insert node@3, item
        
procedure walk node
  if node = nothing
    pass
  else
    walk  node@2
    print node@1
    walk  node@3

procedure main
with root, items, item
  root  := [30, nothing, nothing]
  items := [4, 78, 2, 3, 56, -1, 3]
  for item in items
    insert root, item
  walk  root
  print ""
  print "Tree:"
  print root

The following code implements a bubble sort and uses it to sort a list of integers. 

procedure bubble_sort xs
with swapped, i
  swapped := true
  while swapped
    swapped := false
    for i from 2 to #xs
      if xs@(i - 1) > xs@i
        swap xs, i - 1, i
        swapped := true

procedure swap xs, i, j
with x
  x := xs@i
  xs@i := xs@j
  xs@j := x

procedure main
with xs
  xs := [3, 4, 5, 1, 2]
  bubble_sort xs
  print xs
